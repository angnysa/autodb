package org.angnysa.autodb.metadata.jdbc;

import lombok.SneakyThrows;
import lombok.Value;
import org.angnysa.autodb.metadata.api.Database;
import org.angnysa.autodb.metadata.api.utils.Dumper;
import org.angnysa.autodb.metadata.jdbc.relations.JdbcRelation;

import java.util.Map;

@Value
public class JdbcDatabase implements Database {
	String name;
	String version;
	Map<String, JdbcTable> tables;
	Map<JdbcTable, Map<String, ForeignKey>> foreignKeys;
	Map<JdbcTable, Map<String, JdbcRelation>> relations;

	public JdbcDatabase(String name, String version, Map<String, JdbcTable> tables, Map<JdbcTable, Map<String, ForeignKey>> foreignKeys, Map<JdbcTable, Map<String, JdbcRelation>> relations) {
		this.name = name;
		this.version = version;
		this.tables = Map.copyOf(tables);
		this.foreignKeys = Map.copyOf(foreignKeys);
		this.relations = Map.copyOf(relations);

		tables.values().forEach(t -> t.setDatabase(this));
	}

	@Override
	public Map<JdbcTable, Map<String, JdbcRelation>> getRelations() {
		return relations;
	}

	@SneakyThrows
	@Override
	public String toString() {
		return Dumper.dumpDatabase(this);
	}
}
